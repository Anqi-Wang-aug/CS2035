x1 = 0.01:0.01:20;

    y = [y sin(x1(i))/x1(i)];

function[fmin, fmax, xmin xmax] = search_min_max()
    v = [];
    for i=0.01:0.01:20
        val = sin(i)/i;
        v = [v val];
    end
    fmin = min(v);
    fmax = max(v);
    xmax = find(v==max(v))/100;
    xmin = find(v==min(v))/100;

end
