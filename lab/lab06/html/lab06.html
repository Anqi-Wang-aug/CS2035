q4 rotate 3D

rotate3d allows to change the view point

x=-4.0:0.1:4.0;
y=-3.0:0.1:3.0;
[X,Y] = meshgrid(x,y);
Z = X.^2 - 2*(X.*Y) + 3*Y + 2;
figure
rotate3d on;
surf(X,Y,Z)
axis([-4 4 -3 3 -15 40])
colormap(jet)

q5 surftwo and countertwo

colormap(jet)
x = -2:.1:2; y=x;
[X,Y] = meshgrid(x,y);
Z = X.^2-2*(X.*Y);
figure
rotate3d on;
surf(X,Y,Z+10)
shading interp;
print surftwo.jpg -djpeg
hold on;
figure
axis([-2 2 -2 2 0 25]);
xlabel('X');
ylabel('Y');
contour3(X,Y,Z+10,30,'k')
contour(X,Y,Z+10,30)
print contourtwo.jpg -djpeg
